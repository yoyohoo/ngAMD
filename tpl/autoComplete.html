AutoComplete(Kendo.AutoComplete)

Basic usage:

The elements:
Simple autoComplete: <input id="primaryCities" />
The controller:
$scope.primaryCities = ['Beijing', 'Shanghai', 'Guangzhou', 'Shenzhen'];
$("#primaryCities").kendoAutoComplete({
	dataSource: $scope.primaryCities
});

Advanced usage:

The directive:
simple: <simple-auto-complete></simple-auto-complete>
custom: <custom-autocomplete></custom-autocomplete>
The controller:
Simple:
.directive('simpleAutoComplete', function() {
	return {
		restrict: 'E',
		link: function($scope, $els) {
			$els.find('input').kendoAutoComplete({
				dataSource: $scope.primaryCities
			});
		},
		template: '<input />'
	}
})
Custom:
getJsonp('http://demos.telerik.com/kendo-ui/service/products?callback=angular.callbacks.autoCompleteCallback');
angular.callbacks.autoCompleteCallback = function(response) {
	$scope.customProducts = response;
	$els.find('input').kendoAutoComplete({
		dataTextField: 'ProductName',
		dataSource: $scope.customProducts,
		filter: "startswith",
		placeholder: "Select product...",
		animation: {
			open: {
				effects: "fade:in",
				duration: 500
			}
		}
	});
}
The service:
Custom:
angular.module('uiServices', [])
.factory('getJsonp', ['$http', function($http) {
	return function(url) {
		return $http.jsonp(url);
	}
}])
