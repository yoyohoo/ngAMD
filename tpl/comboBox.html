ComboBox(Kendo.ComboBox)

Basic usage:

The elements:
<select id="cityCombo" placeholder="Select city...">
    <option>beijing</option>
    <option>shanghai</option>
    <option>guangzhou</option>
</select>
The controller:
$('#cityCombo').append('<option>shenzhen</option>');
$('#cityCombo').kendoComboBox();

Advanced usage:

The elements:
<city-combo></city-combo>
The directive:
.directive('cityCombo', function() {
return {
	restrict: 'E',
	scope: {
		type: '=name'
	},
	link: function($scope, $els) {
		setTimeout.call(this, function() {
			$els.find('select[name=cityCombo]').kendoComboBox();
		});
	},
	controller: function($scope) {
		$scope.cities = ['Beijing', 'Shanghai', 'Guangzhou', 'Shenzhen'];
	},
	template: '<select name="cityCombo" placeholder="Select city..."><option ng-repeat="city in cities">{{city}}</option></select>'
}
})

  • restrict: ACE
  • scope: instead of controller or not
  • link: initial the func in directive
  • controller: with the scope
  • template: custom your html
