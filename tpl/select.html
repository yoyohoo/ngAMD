Selects(Bootstrap.Selects)

The elements:
<input type="text" ng-model="newCity">
<a class="btn btn-default" ng-click="AddNewCity()">AddNewCity</a>
<select class="form-control">
  <option ng-repeat="city in primaryCities">{ { city } }</option>
</select>
The controller:
$scope.primaryCities = ['Beijing', 'Shanghai', 'Guangzhou', 'Shenzhen'];
$scope.AddNewCity = function() {
  var a = $scope.primaryCities,
    c = $scope.newCity;
  c && !a.Contains(c) && a.push(c);
}
