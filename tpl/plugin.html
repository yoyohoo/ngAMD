Big alert on the element:

$.fn.extend({
    alert: function(msg, title) {
    	var alert = '<div class="alert-lg alert-suc animated fadeInUp"><strong>' +
    	    title + '</strong> ' + msg + alterClose;
    	if (!this.find('.show-alert').length) this.append(stack);
	return this.find('.show-alert').append(alert);
    }
})

Small alert on the right bottom:

$.fn.extend({
    alert: function(msg, title) {
	var alert = '<div class="alert-sm alert-suc animated fadeInUp"><strong>' +
		title + '</strong>' + msg + alterClose;
	if (!$('body').find('.alert-stack').length) {
		$('body').append(stackSm);
	} else {
		var alerts = $('body').find('.alert-stack .alert-sm');
		alerts.length > 2 && alerts.first().remove();
	}
	return $('body').find('.alert-stack').append(alert);
    }
})
